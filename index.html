                            20: 'Eating',
                            22: 'Resting' }

                            time_now = localtime()
                            hour = time_now.tm_hour

                            for activity_time in sorted(activities.keys()):
                            if hour < activity_time:
                            print activities[activity_time]
                            break
                            else:
                            print 'Unknown, AFK or sleeping!'
